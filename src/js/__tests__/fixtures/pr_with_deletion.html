return fakeCode;
}
},

function treeshakeCode(inputCode) {

input: fakeInput,
onwarn() {},
external: id =>
!id.startsWith('.') &&
!id.startsWith('/') &&

transform(result.code, {compact: true, minified: true, comments: false})
)

originalSize: inputCode.length,
treeshakedSize: result.code.length,
}));
}
